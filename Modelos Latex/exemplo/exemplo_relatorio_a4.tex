\begin{defn}[TFC]
Seja $f:[a,b] \to \R$ uma fun\c c\~ao integr\'avel e $F$ sua primitiva. Ent\~ao

\[
  \int_a^b {f(x)dx}  = F(b) - F(a)
\]
\end{defn}
